DROP TABLE IF EXISTS infectious_cases_normalized;
DROP TABLE IF EXISTS infectious_entities;

CREATE TABLE infectious_entities (
    entity_id INT AUTO_INCREMENT PRIMARY KEY,
    entity VARCHAR(255) NOT NULL,
    code VARCHAR(20),

    CONSTRAINT uq_infectious_entity UNIQUE (entity)
);

INSERT INTO infectious_entities (entity, code)
SELECT DISTINCT
    Entity,
    Code
FROM infectious_cases;

CREATE TABLE infectious_cases_normalized (
    case_id INT AUTO_INCREMENT PRIMARY KEY,
    entity_id INT NOT NULL,
    case_year INT NOT NULL,

    Number_yaws DOUBLE,
    polio_cases DOUBLE,
    cases_guinea_worm DOUBLE,
    Number_rabies DOUBLE,
    Number_malaria DOUBLE,
    Number_hiv DOUBLE,
    Number_tuberculosis DOUBLE,
    Number_smallpox DOUBLE,
    Number_cholera_cases DOUBLE,

    CONSTRAINT fk_infectious_cases_entity
        FOREIGN KEY (entity_id)
        REFERENCES infectious_entities(entity_id),

    CONSTRAINT uq_entity_year
        UNIQUE (entity_id, case_year)
);

INSERT INTO infectious_cases_normalized (
    entity_id,
    case_year,
    Number_yaws,
    polio_cases,
    cases_guinea_worm,
    Number_rabies,
    Number_malaria,
    Number_hiv,
    Number_tuberculosis,
    Number_smallpox,
    Number_cholera_cases
)
SELECT
    e.entity_id,
    ic.`Year`,

    CAST(NULLIF(TRIM(ic.Number_yaws), '') AS DOUBLE),
    CAST(NULLIF(TRIM(ic.polio_cases), '') AS DOUBLE),
    CAST(NULLIF(TRIM(ic.cases_guinea_worm), '') AS DOUBLE),
    CAST(NULLIF(TRIM(ic.Number_rabies), '') AS DOUBLE),
    CAST(NULLIF(TRIM(ic.Number_malaria), '') AS DOUBLE),
    CAST(NULLIF(TRIM(ic.umber_hiv), '') AS DOUBLE),
    CAST(NULLIF(TRIM(ic.Number_tuberculosis), '') AS DOUBLE),
    CAST(NULLIF(TRIM(ic.Number_smallpox), '') AS DOUBLE),
    CAST(NULLIF(TRIM(ic.Number_cholera_cases), '') AS DOUBLE)

FROM infectious_cases ic
JOIN infectious_entities e
    ON ic.Entity = e.entity;
    
SELECT COUNT(*) 
FROM infectious_cases_normalized;
